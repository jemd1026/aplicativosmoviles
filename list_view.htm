class ListViewActivity : AppCompatActivity() {

        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_list_view)

            this,
            productos,
            android.R.layout.simple_list_item_2,
            arrayOf("nombre", "precio"),
            intArrayOf(android.R.id.text1, android.R.id.text2)

        listView.adapter = adapter

import android.widget.Button

class ListViewActivity : AppCompatActivity() {
    
    private lateinit var listView: ListView
    private lateinit var items: Array<String>
    
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_list_view)
        
        listView = findViewById(R.id.listView)
        val btnMostrar = findViewById<Button>(R.id.btnMostrar)
        
        items = resources.getStringArray(R.array.lista_compras)
        
        val adapter = ArrayAdapter(
            this,
            android.R.layout.simple_list_item_checked,
            items
        )
        
        listView.adapter = adapter
        listView.choiceMode = ListView.CHOICE_MODE_MULTIPLE
        
        btnMostrar.setOnClickListener {
            mostrarSeleccionados()
        }
    }
    
    private fun mostrarSeleccionados() {
        val checked = listView.checkedItemPositions
        val seleccionados = mutableListOf<String>()
        
        for (i in 0 until checked.size()) {
            if (checked.valueAt(i)) {
                seleccionados.add(items[checked.keyAt(i)])
            }
        }
        
        if (seleccionados.isEmpty()) {
            Toast.makeText(this, "No hay seleccionados", Toast.LENGTH_SHORT).show()
        } else {
            val mensaje = "Seleccionados:\n${seleccionados.joinToString("\n")}"
            Toast.makeText(this, mensaje, Toast.LENGTH_LONG).show()
        }
    }
}

import androidx.appcompat.app.AlertDialog

class ListViewActivity : AppCompatActivity() {
    
    private lateinit var listView: ListView
    private lateinit var tareas: ArrayList<String>
    private lateinit var adapter: ArrayAdapter<String>
    
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_list_view)
        
        listView = findViewById(R.id.listView)
        
        // Convertir a ArrayList para poder modificar
        val tareasArray = resources.getStringArray(R.array.lista_tareas)
        tareas = ArrayList(tareasArray.toList())
        
        adapter = ArrayAdapter(
            this,
            android.R.layout.simple_list_item_1,
            tareas
        )
        
        listView.adapter = adapter
        
        // Click normal - Ver tarea
        listView.setOnItemClickListener { parent, view, position, id ->
            Toast.makeText(this, "Tarea: ${tareas[position]}", Toast.LENGTH_SHORT).show()
        }
        
        // â­ Long Click - Eliminar tarea
        listView.setOnItemLongClickListener { parent, view, position, id ->
            eliminarTarea(position)
            true // Consumir el evento
        }
    }
    
    private fun eliminarTarea(position: Int) {
        val tarea = tareas[position]
        
        AlertDialog.Builder(this)
            .setTitle("âŒ Eliminar Tarea")
            .setMessage("Â¿Eliminar:\n$tarea?")
            .setPositiveButton("Eliminar") { dialog, which ->
                // Eliminar del ArrayList
                tareas.removeAt(position)
                
                // Notificar cambios al adapter
                adapter.notifyDataSetChanged()
                
                Toast.makeText(this, "âœ… Tarea eliminada", Toast.LENGTH_SHORT).show()
            }
            .setNegativeButton("Cancelar", null)
            .show()
    }
}
